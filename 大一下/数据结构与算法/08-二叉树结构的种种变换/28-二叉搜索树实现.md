```cpp
#include <bits/stdc++.h>
using namespace std;

namespace dsa {

// 红黑树节点：颜色 + key + value + 左右子 + 父
enum Color { RED, BLACK };

template <typename K, typename V>
struct RBNode {
    K key;
    V value;
    Color color;
    RBNode* left;
    RBNode* right;
    RBNode* parent;
    RBNode(K k, V v) : key(k), value(v), color(RED), left(nullptr), right(nullptr), parent(nullptr) {}
};

template <typename K, typename V>
class RBTreeMap {
public:
    RBTreeMap() : root(nullptr), sz(0) {}

    ~RBTreeMap() { clear(root); }

    int size() const { return sz; }
    bool empty() const { return root == nullptr; }

    void put(const K& key, const V& value) {
        RBNode<K,V>* y = nullptr;
        RBNode<K,V>* x = root;
        while (x) {
            y = x;
            if      (key < x->key) x = x->left;
            else if (key > x->key) x = x->right;
            else { x->value = value; return; }   // 已存在则更新
        }
        RBNode<K,V>* z = new RBNode<K,V>(key, value);
        z->parent = y;
        if (!y)                 root = z;
        else if (key < y->key)  y->left  = z;
        else                    y->right = z;
        ++sz;
        insertFixup(z);
    }

    bool get(const K& key, V& value) const {
        RBNode<K,V>* n = find(key);
        if (!n) return false;
        value = n->value;
        return true;
    }

    bool containsKey(const K& key) const { return find(key) != nullptr; }

    void remove(const K& key) {
        RBNode<K,V>* z = find(key);
        if (!z) return;
        deleteNode(z);
        --sz;
    }

    K minKey() const {
        if (!root) throw runtime_error("empty");
        RBNode<K,V>* n = root;
        while (n->left) n = n->left;
        return n->key;
    }

    K maxKey() const {
        if (!root) throw runtime_error("empty");
        RBNode<K,V>* n = root;
        while (n->right) n = n->right;
        return n->key;
    }

    // floorKey：返回 ≤ key 的最大键
    K floorKey(const K& key) const {
        RBNode<K,V>* n = root, *ans = nullptr;
        while (n) {
            if (n->key <= key) { ans = n; n = n->right; }
            else                n = n->left;
        }
        if (!ans) throw runtime_error("no floor");
        return ans->key;
    }

    // ceilingKey：返回 ≥ key 的最小键
    K ceilingKey(const K& key) const {
        RBNode<K,V>* n = root, *ans = nullptr;
        while (n) {
            if (n->key >= key) { ans = n; n = n->left; }
            else                n = n->right;
        }
        if (!ans) throw runtime_error("no ceiling");
        return ans->key;
    }

    vector<K> keys() const {
        vector<K> out;
        inorder(root, out);
        return out;
    }

private:
    RBNode<K,V>* root;
    int sz;

    RBNode<K,V>* find(const K& key) const {
        RBNode<K,V>* n = root;
        while (n) {
            if      (key < n->key) n = n->left;
            else if (key > n->key) n = n->right;
            else                   return n;
        }
        return nullptr;
    }

    void inorder(RBNode<K,V>* n, vector<K>& out) const {
        if (!n) return;
        inorder(n->left, out);
        out.push_back(n->key);
        inorder(n->right, out);
    }

    void clear(RBNode<K,V>* n) {
        if (!n) return;
        clear(n->left);
        clear(n->right);
        delete n;
    }

    // 左旋
    void rotateLeft(RBNode<K,V>* x) {
        RBNode<K,V>* y = x->right;
        x->right = y->left;
        if (y->left) y->left->parent = x;
        y->parent = x->parent;
        if (!x->parent)        root = y;
        else if (x == x->parent->left)  x->parent->left  = y;
        else                            x->parent->right = y;
        y->left = x;
        x->parent = y;
    }

    // 右旋
    void rotateRight(RBNode<K,V>* x) {
        RBNode<K,V>* y = x->left;
        x->left = y->right;
        if (y->right) y->right->parent = x;
        y->parent = x->parent;
        if (!x->parent)        root = y;
        else if (x == x->parent->right) x->parent->right = y;
        else                            x->parent->left  = y;
        y->right = x;
        x->parent = y;
    }

    // 插入后修复红黑树性质
    void insertFixup(RBNode<K,V>* z) {
        while (z->parent && z->parent->color == RED) {
            RBNode<K,V>* gp = z->parent->parent;
            if (z->parent == gp->left) {
                RBNode<K,V>* u = gp->right;
                if (u && u->color == RED) {           // Case 1: 叔红
                    z->parent->color = BLACK;
                    u->color = BLACK;
                    gp->color = RED;
                    z = gp;
                } else {
                    if (z == z->parent->right) {     // Case 2: 叔黑 + z 是右子
                        z = z->parent;
                        rotateLeft(z);
                    }
                    // Case 3: 叔黑 + z 是左子
                    z->parent->color = BLACK;
                    gp->color = RED;
                    rotateRight(gp);
                }
            } else {                                  // 对称：z->parent 是右子
                RBNode<K,V>* u = gp->left;
                if (u && u->color == RED) {
                    z->parent->color = BLACK;
                    u->color = BLACK;
                    gp->color = RED;
                    z = gp;
                } else {
                    if (z == z->parent->left) {
                        z = z->parent;
                        rotateRight(z);
                    }
                    z->parent->color = BLACK;
                    gp->color = RED;
                    rotateLeft(gp);
                }
            }
        }
        root->color = BLACK;
    }

    // 用 v 替代 u（u 的父节点的 left/right 指向 v，v 的 parent 指向 u 的父节点）
    void transplant(RBNode<K,V>* u, RBNode<K,V>* v) {
        if (!u->parent)            root = v;
        else if (u == u->parent->left)  u->parent->left  = v;
        else                              u->parent->right = v;
        if (v) v->parent = u->parent;
    }

    // 子树最小节点
    RBNode<K,V>* minNode(RBNode<K,V>* x) const {
        while (x->left) x = x->left;
        return x;
    }

    void deleteNode(RBNode<K,V>* z) {
        RBNode<K,V>* y = z;
        Color yOriginalColor = y->color;
        RBNode<K,V>* x = nullptr;
        RBNode<K,V>* xParent = nullptr;

        if (!z->left) {
            x = z->right;
            xParent = z->parent;
            transplant(z, z->right);
        } else if (!z->right) {
            x = z->left;
            xParent = z->parent;
            transplant(z, z->left);
        } else {
            y = minNode(z->right);
            yOriginalColor = y->color;
            x = y->right;
            if (y->parent == z) {
                xParent = y;
            } else {
                transplant(y, y->right);
                y->right = z->right;
                y->right->parent = y;
                xParent = y->parent;
            }
            transplant(z, y);
            y->left = z->left;
            y->left->parent = y;
            y->color = z->color;
        }
        delete z;
        if (yOriginalColor == BLACK) deleteFixup(x, xParent);
    }

    void deleteFixup(RBNode<K,V>* x, RBNode<K,V>* xParent) {
        while ((!x || x->color == BLACK) && x != root) {
            if (x == xParent->left) {
                RBNode<K,V>* w = xParent->right;
                if (w && w->color == RED) {                        // Case 1
                    w->color = BLACK;
                    xParent->color = RED;
                    rotateLeft(xParent);
                    w = xParent->right;
                }
                if ((!w->left  || w->left->color  == BLACK) &&
                    (!w->right || w->right->color == BLACK)) {     // Case 2
                    if (w) w->color = RED;
                    x = xParent;
                    xParent = x->parent;
                } else {
                    if (!w->right || w->right->color == BLACK) {  // Case 3
                        if (w->left) w->left->color = BLACK;
                        w->color = RED;
                        rotateRight(w);
                        w = xParent->right;
                    }
                    // Case 4
                    w->color = xParent->color;
                    xParent->color = BLACK;
                    if (w->right) w->right->color = BLACK;
                    rotateLeft(xParent);
                    x = root;
                    break;
                }
            } else {  // 对称分支
                RBNode<K,V>* w = xParent->left;
                if (w && w->color == RED) {
                    w->color = BLACK;
                    xParent->color = RED;
                    rotateRight(xParent);
                    w = xParent->left;
                }
                if ((!w->right || w->right->color == BLACK) &&
                    (!w->left  || w->left->color  == BLACK)) {
                    if (w) w->color = RED;
                    x = xParent;
                    xParent = x->parent;
                } else {
                    if (!w->left || w->left->color == BLACK) {
                        if (w->right) w->right->color = BLACK;
                        w->color = RED;
                        rotateLeft(w);
                        w = xParent->left;
                    }
                    w->color = xParent->color;
                    xParent->color = BLACK;
                    if (w->left) w->left->color = BLACK;
                    rotateRight(xParent);
                    x = root;
                    break;
                }
            }
        }
        if (x) x->color = BLACK;
    }
};

} // namespace dsa

int main() {
    using namespace dsa;
    RBTreeMap<string, int> m;
    m.put("apple",  1);
    m.put("banana", 2);
    m.put("cherry", 3);
    m.put("date",   4);

    cout << "size: " << m.size() << endl;             // 4
    cout << "min: "  << m.minKey() << endl;           // apple
    cout << "max: "  << m.maxKey() << endl;           // date
    cout << "floor(cherry): " << m.floorKey("cherry") << endl;     // cherry
    cout << "floor(cocoa): " << m.floorKey("cocoa") << endl;       // cherry
    cout << "ceiling(cocoa): " << m.ceilingKey("cocoa") << endl;   // date

    int v;
    m.get("banana", v);  cout << "banana=" << v << endl;  // 2
    m.remove("banana");
    cout << "contains banana: " << m.containsKey("banana") << endl; // 0

    cout << "keys: ";
    for (auto& k : m.keys()) cout << k << " ";  // apple cherry date
    cout << endl;
    return 0;
}
```

## 3. TreeSet：TreeMap 的简单封装

> TreeSet 就是「只用 key、不用 value」的 TreeMap，所以实现就是 TreeMap 的一层封装。

```cpp
template <typename K>
class TreeSet {
private:
    RBTreeMap<K, bool> m;  // value 没有意义
public:
    void add(const K& key)   { m.put(key, true); }
    bool contains(const K& k) { return m.containsKey(k); }
    void remove(const K& k)  { m.remove(k); }
    int  size() const        { return m.size(); }
    bool empty() const       { return m.empty(); }
    K    min() const         { return m.minKey(); }
    K    max() const         { return m.maxKey(); }
    vector<K> elements() const { return m.keys(); }
};
```

    K key;
    V value;
    Entry<K,V> left;

    // ...

### 4.3 关键操作

- `put(key, value)`：BST 插入 + 红黑树 fixAfterInsertion。
- `get(key)`：BST 查找。
- `remove(key)`：BST 删除 + 红黑树 fixAfterDeletion。
- `firstKey()` / `lastKey()`：返回最左/最右节点。
- `floorEntry(key)` / `ceilingEntry(key)`：沿 BST 路径下降。
- `subMap(from, to)`：返回子树视图，惰性计算。

> [!tip] 关键细节
> Java TreeMap 的 `key` **必须可比较**（实现 `Comparable<K>` 接口），否则需要在构造时传入 `Comparator<K>`。这是红黑树做 key 比较的前提。

## 5. C++ `std::map` 内部结构


### 5.1 头文件与类模板

```cpp
#include <map>

          class Compare = std::less<Key>,

```

### 5.2 底层数据结构

C++ `std::map` / `std::set` / `std::multimap` / `std::multiset` 这四种容器的底层都使用同一个组件：**`_Rb_tree`（红黑树）**。

```text
std::map<K, V>      ── 适配器 ──>  _Rb_tree<K, V, ...>
std::set<K>         ── 适配器 ──>  _Rb_tree<K, V, ...>   (V 永远是 __identity)
std::multimap<K, V> ── 适配器 ──>  _Rb_tree<K, V, ...>   (允许 key 重复)
std::multiset<K>    ── 适配器 ──>  _Rb_tree<K, V, ...>   (允许 key 重复)
```

### 5.3 关键源码（libstdc++，简化）

```cpp
template <typename _Key, typename _Tp, typename _Compare = std::less<_Key>,
          typename _Alloc = std::allocator<std::pair<const _Key, _Tp>>>
class map {
public:
    typedef _Rb_tree< _Key, std::pair<const _Key, _Tp>,
                      _Select1st<std::pair<const _Key, _Tp>>,
                      _Compare, _Alloc > _Rep_type;
    _Rep_type _M_t;  // 内部的红黑树
    // ...
    std::pair<iterator, bool> insert(const value_type& x) {
        return _M_t._M_insert_unique(x);
    }
    iterator find(const key_type& k) { return _M_t.find(k); }
    size_type erase(const key_type& k) { return _M_t.erase(k); }
};
```

## 6. TreeMap vs HashMap 怎么选

| 维度 | TreeMap | HashMap |
|------|---------|---------|
| 底层 | 红黑树 | 哈希表 |
| 单次操作复杂度 | **O(log N)** | O(1) 平均 |
| key 是否有序 | ✅ 有序 | ❌ 无序 |
| 实现复杂度 | 高（红黑树旋转） | 中（哈希冲突） |
| 适用场景 | 范围查询 / 排名 / 按序遍历 | 仅需 O(1) 查找 |
| 内存占用 | 每个节点多 1 个 color + parent 指针 | 数组 + 链表 / 红黑树 |
| 选谁？ | **需要有序 / 范围查询** | **只要快** |

> [!tip] 一句话选择指南
> - **需要 key 有序** → TreeMap
> - **只要快** → HashMap
> - **key 是整数 + 需要 O(1) + 范围查询** → 考虑 [[45-堆排序|树状数组 (BIT)]]

## 7. 应用场景

- **Java `TreeMap` / C++ `std::map` / Python `SortedDict`**：所有需要「按 key 排序的 map」场景
- **滑动窗口 + 范围查询**：用 TreeSet 维护窗口，O(log N) 插入/删除 + 范围查询
- **日程表 / 时间线**：TreeMap 按时间戳排序，O(log N) 插入新事件，O(log N) 查询重叠
- **排行榜**：TreeMap 存「分数 → 玩家集合」，按分数排序
- **数据库索引**：B+ 树（红黑树的多路推广）就是数据库索引的底层

## 8. TreeMap 实战代码片段

> 下面是 5 个最常见的 TreeMap 实战场景，每个都给一段可直接复用的 C++ / Java 风格代码。

### 8.1 滑动窗口中的最接近元素

题目：维护一个大小为 K 的滑动窗口，O(log K) 查询窗口内最接近 `x` 的元素。

```cpp
#include <set>
#include <vector>
#include <cstdlib>
using namespace std;

vector<int> closestInWindow(const vector<int>& nums, int k, int x) {
    multiset<int> window;          // multiset 是基于红黑树的有序多重集
    vector<int> res;
    for (int i = 0; i < (int)nums.size(); ++i) {
        window.insert(nums[i]);
        if ((int)window.size() > k) window.erase(window.find(nums[i - k]));
        if ((int)window.size() == k) {
            // 找最接近 x 的元素：用 lower_bound 找第一个 ≥ x 的，再和前一个比较
            auto it = window.lower_bound(x);
            int best = *it;
            if (it != window.begin()) {
                --it;
                if (abs(*it - x) < abs(best - x)) best = *it;
            }
            res.push_back(best);
        }
    }
    return res;
}
```

### 8.2 区间合并 / 区间调度

题目：给定若干区间 `[[s1, e1], [s2, e2], ...]`，合并所有重叠的区间。

```cpp
#include <map>
#include <vector>
using namespace std;

vector<pair<int,int>> mergeIntervals(vector<pair<int,int>>& intervals) {
    map<int, int> m;   // start → end
    for (auto& [s, e] : intervals) {
        auto it = m.lower_bound(s);
        // 检查左边的区间是否重叠
        if (it != m.begin()) {
            --it;
            if (it->second < s) ++it;
        }
        int newStart = s, newEnd = e;
        // 合并所有重叠的区间
        while (it != m.end() && it->first <= newEnd) {
            newStart = min(newStart, it->first);
            newEnd  = max(newEnd,  it->second);
            it = m.erase(it);
        }
        m[newStart] = newEnd;
    }
    return vector<pair<int,int>>(m.begin(), m.end());
}
```

### 8.3 频率排序

题目：把数组按「出现频率从高到低、相同频率按值升序」排序。

```cpp
#include <map>
#include <set>
#include <vector>
using namespace std;

vector<int> frequencySort(const vector<int>& nums) {
    map<int, int> freq;
    for (int x : nums) ++freq[x];

    // 用 set 排序：(频率降序, 值升序) → 自定义比较器
    set<pair<int,int>, greater<pair<int,int>>> order;  // 默认 less，这里改 greater
    // 注意：multiset 支持按 (freq, value) 排序，自动按字典序
    multiset<pair<int,int>> order2;   // (freq, value)，默认升序
    for (auto& [v, f] : freq) order2.insert({f, v});

    vector<int> res;
    for (auto& [f, v] : order2) {
        for (int i = 0; i < f; ++i) res.push_back(v);
    }
    return res;
}
```

### 8.4 维护数据流的中位数

题目：实时插入数据，用 O(log N) 维护中位数。

```cpp
#include <queue>
using namespace std;

class MedianFinder {
    priority_queue<int> maxHeap;                          // 存较小的一半，大顶堆
    priority_queue<int, vector<int>, greater<int>> minHeap; // 存较大的一半，小顶堆
public:
    void addNum(int num) {
        if (maxHeap.empty() || num <= maxHeap.top()) maxHeap.push(num);
        else                                          minHeap.push(num);
        // 平衡：两个堆大小差 ≤ 1
        if ((int)maxHeap.size() > (int)minHeap.size() + 1) {
            minHeap.push(maxHeap.top()); maxHeap.pop();
        } else if ((int)minHeap.size() > (int)maxHeap.size()) {
            maxHeap.push(minHeap.top()); minHeap.pop();
        }
    }
    double findMedian() {
        if (maxHeap.size() == minHeap.size()) {
            return (maxHeap.top() + minHeap.top()) / 2.0;
        }
        return maxHeap.top();
    }
};
```

> [!tip] 为什么用 priority_queue 而非 TreeMap？
> 中位数问题需要 **动态维护最值**，[[22-二叉堆实现|堆]]（priority_queue）是更直接的工具；但如果你需要「中位数 ± 任意 k 的元素」，TreeMap / TreeSet 才是正解。

### 8.5 找出第 K 小的数（数据流版）

题目：维护一个数据流，支持 `add(num)` 和 `kthSmallest(k)`。

```cpp
#include <set>
using namespace std;

class KthSmallest {
    multiset<int> s;            // 多重集
    multiset<int> topK;         // 始终保持 topK 的大小 = K
    int k;
public:
    KthSmallest(int _k) : k(_k) {}

    void add(int num) {
        s.insert(num);
        topK.insert(num);
        if ((int)topK.size() > k) {
            // topK 移除最大的那个
            auto it = prev(topK.end());
            topK.erase(it);
        }
    }

    int kthSmallest() {
        if (topK.size() < k) return -1;       // 不足 K 个
        return *prev(topK.end());             // 最大的就是第 K 小的
    }
};
```

## 9. TreeMap 选型备忘单

> 下面这张「什么时候用什么」决策表帮你快速选型。

| 需求 | 推荐数据结构 | 原因 |
|------|------------|------|
| 维护有序 key-value | TreeMap / std::map | 天然有序，O(log N) |
| 维护有序 key 集合 | TreeSet / std::set | 同上 |
| 只需要 O(1) 查找 | HashMap / unordered_map | 平均 O(1) |
| key 是字符串 + 前缀查询 | Trie | 天然支持前缀 |
| 维护第 K 小 | TreeMap + 子树 size 字段 | O(log N) |
| 维护中位数 | 两个堆 | 维护最值更快 |
| 滑动窗口 + 范围查询 | TreeSet | 排序 + 范围 |
| 图的连通性 | Union Find | O(1) 合并 |
| 范围加法 + 范围求和 | 线段树 / 树状数组 | 区间聚合 |
| 大数据 + 顺序 IO | B+ 树 | 数据库索引 |

> [!tip] 跨语言对照
> - **C++**：`std::map` / `std::set`（红黑树），`std::unordered_map` / `std::unordered_set`（哈希表）
> - **Java**：`TreeMap` / `TreeSet`（红黑树），`HashMap` / `HashSet`（哈希表）
> - **Python**：`SortedDict`（第三方库，红黑树），`dict`（哈希表）
> - **Go**：`map`（哈希表，无内置红黑树，但有 `container/list`）
> - **Rust**：`BTreeMap`（B 树，类红黑树），`HashMap`（哈希表）
