# 2.62
```c
int int_shifts_are_arithmetic()
{
	int num = -1;
	return !(num ^ (num >> 1));
}

## B

int int_size_is_32()
{
	int set_msb = 1 << 31;
	int beyond_msb = set_msb << 1;
	return set_msb && !beyond_msb;
}
```

int int_size_is_32()
{
	int set_msb = 1 << 15;
	set_msb <<= 15;
	set_msb <<= 1;
	
	int beyond_msb = set_msb << 1;
	
	return set_msb && !beyond_msb;
}
