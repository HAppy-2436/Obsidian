int int_shifts_are_arithmetic()
{
	int num = -1;
	return !(num ^ (num >> 1));
}

int signed_high_prod(int x, int y)
{
	unsigned u_high = unsigned_high_prod((unsigned) x, (unsigned) y);
	int correction = (x >> 31) & y;
	correction += (y >> 31) & x;
	return (int) (u_high - correction);
}
